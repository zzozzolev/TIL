### my solution
```sql
WITH ECONOMY_BOOK AS (
    SELECT BOOK_ID, AUTHOR_ID, PUBLISHED_DATE FROM BOOK WHERE CATEGORY = '경제'
)

SELECT BOOK_ID, AUTHOR_NAME, DATE_FORMAT(PUBLISHED_DATE, '%Y-%m-%d') AS PUBLISHED_DATE
FROM ECONOMY_BOOK b JOIN AUTHOR a ON b.AUTHOR_ID = a.AUTHOR_ID
ORDER BY PUBLISHED_DATE
```
