### my solution
```sql
WITH july_sum AS (
    SELECT FLAVOR, SUM(TOTAL_ORDER) AS TOTAL
    FROM JULY
    GROUP BY FLAVOR
), first_half_sum AS (
    SELECT FLAVOR, SUM(TOTAL_ORDER) AS TOTAL
    FROM FIRST_HALF
    GROUP BY FLAVOR
)

SELECT DISTINCT(IFNULL(JULY_FLAVOR, HALF_FLAVOR)) AS FLAVOR
FROM (
    SELECT *
    FROM (SELECT FLAVOR AS JULY_FLAVOR, TOTAL AS JULY_TOTAL FROM july_sum) t1
    LEFT JOIN (SELECT FLAVOR AS HALF_FLAVOR, TOTAL AS HALF_TOTAL FROM first_half_sum) t2
    ON t1.JULY_FLAVOR = t2.HALF_FLAVOR
    UNION
    SELECT *
    FROM (SELECT FLAVOR AS HALF_FLAVOR, TOTAL AS HALF_TOTAL FROM first_half_sum) t4
    LEFT JOIN (SELECT FLAVOR AS JULY_FLAVOR, TOTAL AS JULY_TOTAL FROM july_sum) t3
    ON t3.JULY_FLAVOR = t4.HALF_FLAVOR
) t
ORDER BY JULY_TOTAL + HALF_TOTAL DESC
LIMIT 3;
```
