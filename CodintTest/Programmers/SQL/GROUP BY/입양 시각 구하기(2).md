### my solution
```sql
WITH RECURSIVE HOUR_SEQ(HOUR) AS (
    SELECT 0
    UNION ALL
    SELECT HOUR + 1
    FROM HOUR_SEQ
    WHERE HOUR < 23
), HOUR_GROUP_BY AS (
    SELECT HOUR(DATETIME) AS HOUR, COUNT(*) AS COUNT
    FROM ANIMAL_OUTS
    GROUP BY HOUR
)

SELECT s.HOUR AS HOUR, IF(g.COUNT IS NULL, 0, g.COUNT) AS COUNT
FROM HOUR_SEQ s LEFT JOIN HOUR_GROUP_BY g ON s.HOUR = g.HOUR
ORDER BY HOUR
```

### other solution
```sql
WITH RECURSIVE GET_HOUR (HOUR)
  AS (SELECT 0
       UNION ALL
      SELECT HOUR+1
        FROM GET_HOUR
       WHERE HOUR < 23
     )
SELECT G.HOUR, IFNULL(COUNT(ANIMAL_ID),'0')
FROM GET_HOUR G
LEFT OUTER JOIN ANIMAL_OUTS A ON G.HOUR = HOUR(DATETIME)
GROUP BY 1
ORDER BY 1
```
